def pbExportarPokedexJSON
  require 'json'
  pokedex = []
  mapa_numeros = {}
  indice_regional = 1

  # 1. Obtenemos el orden original del motor (el orden de los PBS)
  # GameData::Species.keys nos da los símbolos en el orden correcto
  orden_pbs = GameData::Species.keys

  # 2. Recolectamos todas las especies en una lista plana
  todas_las_especies = []
  GameData::Species.each { |s| todas_las_especies << s }

  # 3. ORDENAMIENTO MAESTRO:
  # - Primero: Buscamos la posición de la especie BASE en el orden del PBS
  # - Segundo: Ordenamos por número de forma (0, 1, 2...) para que las megas sigan a su base
  especies_ordenadas = todas_las_especies.sort_by do |s| 
    [orden_pbs.index(s.species), s.form] 
  end

  especies_ordenadas.each do |s|
    next if s.name.nil? || s.name == ""
    
    # Lógica de número regional (Venusaur y Mega Venusaur comparten el mismo)
    if s.form == 0
      numero_pokedex = indice_regional
      mapa_numeros[s.species] = numero_pokedex
      indice_regional += 1
    else
      numero_pokedex = mapa_numeros[s.species] || 0
    end

    pokedex << {
      "numero" => numero_pokedex,
      "id" => s.id.to_s,
      "nombre" => s.name,
      "form_name" => s.form_name || "",
      "es_forma" => s.form > 0,
      "generacion" => s.generation || 0,
      "tipos" => s.types.map { |t| t.to_s },
      "stats_base" => {
        "hp"  => s.base_stats[:HP],
        "atq" => s.base_stats[:ATTACK],
        "def" => s.base_stats[:DEFENSE],
        "spa" => s.base_stats[:SPECIAL_ATTACK],
        "spd" => s.base_stats[:SPECIAL_DEFENSE],
        "vel" => s.base_stats[:SPEED]
      },
      "física" => {
        "altura" => s.height / 10.0,
        "peso" => s.weight / 10.0
      },
      "habilidades" => s.abilities.map { |a| a.to_s },
      "habilidad_oculta" => s.hidden_abilities.map { |a| a.to_s },
      "descripcion" => s.pokedex_entry
    }
  end

  File.open("valerion_data.json", "w") { |f| f.write(JSON.pretty_generate(pokedex)) }
  pbMessage(_INTL("JSON exportado en orden Nacional/PBS con #{indice_regional-1} especies."))
end
